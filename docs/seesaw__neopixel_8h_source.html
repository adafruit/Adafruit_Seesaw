28 // The order of primary colors in the NeoPixel data stream can vary
29 // among device types, manufacturers and even different revisions of
30 // the same item. The third parameter to the seesaw_NeoPixel
31 // constructor encodes the per-pixel byte offsets of the red, green
32 // and blue primaries (plus white, if present) in the data stream --
33 // the following #defines provide an easier-to-use named version for
34 // each permutation. e.g. NEO_GRB indicates a NeoPixel-compatible
35 // device expecting three bytes per pixel, with the first byte
36 // containing the green value, second containing red and third
37 // containing blue. The in-memory representation of a chain of
38 // NeoPixels is the same as the data-stream order; no re-ordering of
39 // bytes is required when issuing data to the chain.
40 
41 // Bits 5,4 of this value are the offset (0-3) from the first byte of
42 // a pixel to the location of the red color byte. Bits 3,2 are the
43 // green offset and 1,0 are the blue offset. If it is an RGBW-type
44 // device (supporting a white primary in addition to R,G,B), bits 7,6
45 // are the offset to the white byte...otherwise, bits 7,6 are set to
46 // the same value as 5,4 (red) to indicate an RGB (not RGBW) device.
47 // i.e. binary representation:
48 // 0bWWRRGGBB for RGBW devices
49 // 0bRRRRGGBB for RGB
50 
51 // RGB NeoPixel permutations; white and red offsets are always same
52 // Offset: W R G B
53 #define NEO_RGB ((0 << 6) | (0 << 4) | (1 << 2) | (2))
54 #define NEO_RBG ((0 << 6) | (0 << 4) | (2 << 2) | (1))
55 #define NEO_GRB ((1 << 6) | (1 << 4) | (0 << 2) | (2))
56 #define NEO_GBR ((2 << 6) | (2 << 4) | (0 << 2) | (1))
57 #define NEO_BRG ((1 << 6) | (1 << 4) | (2 << 2) | (0))
58 #define NEO_BGR ((2 << 6) | (2 << 4) | (1 << 2) | (0))
59 
60 // RGBW NeoPixel permutations; all 4 offsets are distinct
61 // Offset: W R G B
62 #define NEO_WRGB ((0 << 6) | (1 << 4) | (2 << 2) | (3))
63 #define NEO_WRBG ((0 << 6) | (1 << 4) | (3 << 2) | (2))
64 #define NEO_WGRB ((0 << 6) | (2 << 4) | (1 << 2) | (3))
65 #define NEO_WGBR ((0 << 6) | (3 << 4) | (1 << 2) | (2))
66 #define NEO_WBRG ((0 << 6) | (2 << 4) | (3 << 2) | (1))
67 #define NEO_WBGR ((0 << 6) | (3 << 4) | (2 << 2) | (1))
68 
69 #define NEO_RWGB ((1 << 6) | (0 << 4) | (2 << 2) | (3))
70 #define NEO_RWBG ((1 << 6) | (0 << 4) | (3 << 2) | (2))
71 #define NEO_RGWB ((2 << 6) | (0 << 4) | (1 << 2) | (3))
72 #define NEO_RGBW ((3 << 6) | (0 << 4) | (1 << 2) | (2))
73 #define NEO_RBWG ((2 << 6) | (0 << 4) | (3 << 2) | (1))
74 #define NEO_RBGW ((3 << 6) | (0 << 4) | (2 << 2) | (1))
75 
76 #define NEO_GWRB ((1 << 6) | (2 << 4) | (0 << 2) | (3))
77 #define NEO_GWBR ((1 << 6) | (3 << 4) | (0 << 2) | (2))
78 #define NEO_GRWB ((2 << 6) | (1 << 4) | (0 << 2) | (3))
79 #define NEO_GRBW ((3 << 6) | (1 << 4) | (0 << 2) | (2))
80 #define NEO_GBWR ((2 << 6) | (3 << 4) | (0 << 2) | (1))
81 #define NEO_GBRW ((3 << 6) | (2 << 4) | (0 << 2) | (1))
82 
83 #define NEO_BWRG ((1 << 6) | (2 << 4) | (3 << 2) | (0))
84 #define NEO_BWGR ((1 << 6) | (3 << 4) | (2 << 2) | (0))
85 #define NEO_BRWG ((2 << 6) | (1 << 4) | (3 << 2) | (0))
86 #define NEO_BRGW ((3 << 6) | (1 << 4) | (2 << 2) | (0))
87 #define NEO_BGWR ((2 << 6) | (3 << 4) | (1 << 2) | (0))
88 #define NEO_BGRW ((3 << 6) | (2 << 4) | (1 << 2) | (0))
